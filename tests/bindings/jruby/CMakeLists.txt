# Build java tests
set(JRUBY_TEST_TARGET TestJRuby)
message(STATUS "Configuring ${JRUBY_TEST_TARGET}")

add_test(NAME ${JRUBY_TEST_TARGET} 
    COMMAND ${JRUBY_EXECUTABLE} ${CMAKE_CURRENT_LIST_DIR}/${JRUBY_TEST_TARGET}.rb $<TARGET_FILE:${SERVER_TARGET}> 
)
set_tests_properties(${JRUBY_TEST_TARGET} PROPERTIES ENVIRONMENT JRUBY_OPTS="-J-Djava.library.path=${CMAKE_RUNTIME_OUTPUT_DIRECTORY} -I ${CMAKE_RUNTIME_OUTPUT_DIRECTORY}")
