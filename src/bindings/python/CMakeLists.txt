set(PYTHON_INTERFACES
    py_showtime.i
)

SET_SOURCE_FILES_PROPERTIES(${PYTHON_INTERFACES} PROPERTIES CPLUSPLUS ON)

FIND_PACKAGE(PythonLibs 3.6 REQUIRED)
INCLUDE_DIRECTORIES(${PYTHON_INCLUDE_PATH} ${BASEPATH})
include (${SWIG_USE_FILE})
set(CMAKE_SWIG_OUTDIR ${CMAKE_CURRENT_BINARY_DIR})

SWIG_ADD_LIBRARY(showtime_py LANGUAGE python SOURCES ${PYTHON_INTERFACES} ${ZST_HEADERS})
set_target_properties(${SWIG_MODULE_${showtime_py}_REAL_NAME} PROPERTIES LIBRARY_OUTPUT_DIRECTORY ${CMAKE_SWIG_OUTDIR})

SWIG_LINK_LIBRARIES(showtime_py 
	${PROJECT_NAME}
	${ZeroMQ_LIBRARIES} 
	${PYTHON_LIBRARIES}
	${CZMQ_LIBRARIES}
	${MSGPACK_LIBRARIES}
)
